local function OnActorDie(e)
    local res, ret = pcall(function ()
        local id = e.eventobjid
        local killer_id = e.toobjid
        local actor_id = e.actorid or 0
        G_EntityManager:OnActorDie(id, killer_id, actor_id)
    end)
    if not res then
        error(ret, 2)
    end
    assert(res, ret)
end
